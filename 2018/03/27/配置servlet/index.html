配置servlet

发表于 2018-03-27

摘要:

- 经常需要配置servlet环境，但是东西太多总是记不住，做一下记录，完整项目:https://github.com/dee1996/Servlet目录下LibrarySystem

正文:

- 配置web.xml
    <?xml version="1.0" encoding="UTF-8"?>
    <web-app version="3.0" 
        xmlns="http://java.sun.com/xml/ns/javaee" 
        xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" 
        xsi:schemaLocation="http://java.sun.com/xml/ns/javaee 
        http://java.sun.com/xml/ns/javaee/web-app_3_0.xsd">
        <!-- 这里是为了转码，将所有的字符编码转为utf-8 -->
        <filter>
        <filter-name>RequestFilter</filter-name>
        <filter-class>cn.landis.filter.RequestFilter</filter-class>
        <init-param>
        <param-name>charset</param-name>
        <param-value>utf-8</param-value>
        </init-param>
        </filter>
        <filter-mapping>
        <filter-name>RequestFilter</filter-name>
        <url-pattern>/*</url-pattern>
        </filter-mapping>

      <welcome-file-list>
        <welcome-file>index.jsp</welcome-file>
      </welcome-file-list>

    <servlet>
    <servlet-name>action</servlet-name>
    <servlet-class>cn.landis.servlet.Servlet</servlet-class>
    </servlet>    


    <servlet-mapping>
    <servlet-name>action</servlet-name>
    <url-pattern>*.do</url-pattern>
    </servlet-mapping>

    </web-app>
